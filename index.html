Reglas correctas de Firestore

Publica estas reglas para permitir escritura a cualquier usuario autenticado.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    match /products/{productId} {
      allow read: if true;
      allow create, update, delete: if signedIn();
    }

    match /settings/{docId} {
      allow read: if docId == "public";
      allow create, update, delete: if docId == "public" && signedIn();
    }
  }
}
